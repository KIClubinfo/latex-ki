	\begin{frame}[fragile=singleslide]
		\frametitle{Solution}

			\begin{verbatim}
			On a ainsi,
			\begin{equation}
			\forall i \in [1,N], \ \ \ \ \pi_i
			\propto e^{-\frac{E_i}{kT}}
			\end{equation}

			et par normalisation de la somme des probabilités,
			on trouve finalement un
			coefficient de proportionnalité valant $1/Z$, avec
